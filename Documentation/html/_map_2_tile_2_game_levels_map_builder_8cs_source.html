325  {

328  // dictionary of levels, useful for fast checking level tiles
329  Dictionary<int, string> levelTiles = new Dictionary<int, string>();
330 
331  foreach (var gameLevelReference in map.Levels)
332  {
333  levelTiles.Add(gameLevelReference.Hash, gameLevelReference.Id);
334  }
335 
336  // create root tiles first
337  foreach (var rootTile in map.Roots)
338  {
339  T tile = null;
340  tile = _factory.NewInstance<T>();
341  RegisterTile(tile, rootTile.Key);
342  currentRoots.Add(rootTile.Key);
343  // set transform
344  tile.transform.position = rootTile.Position;
345  tile.transform.rotation = Quaternion.Euler(rootTile.Rotation);
346  // register root within the tile map
347  if (_map.Roots == null) _map.Roots = new List<RootTile>();
348 
349  _map.Roots.Add(rootTile);
350  tile.Controller.Tile.Root = true;
351 
352  tile.name = $"(Root) {tile.Controller.Tile.Index}";
353 
354  // check if root is a level tile
355  if (levelTiles.TryGetValue(rootTile.Key, out var levelId))
356  {
357  tile.name = $"'{levelId}' {tile.name}";
358  tile.LevelId = levelId;
359  tile.ConvertToLevelTile();
360  }
361  }
362 
363  var hasNewRootAvailable = true;
364 
365  // Now execute neighbor insertion logic
366  while (hasNewRootAvailable)
367  {
368  var newRoots = new List<int>();
369  hasNewRootAvailable = false;
370 
371  // now create all the tiles related to these roots
372  foreach (var root in currentRoots)
373  {
374  var related = map.Instructions.Where(x => x.Root == root).ToList();
375  var anyRelated = related.Any();
376  // ensure next loop since there is at least another tile parent
377  hasNewRootAvailable |= anyRelated;
378 
379  // no need to iterate if this tile is a leaf
380  if (!anyRelated) continue;
381 
382  foreach (var instruction in related)
383  {
384  var rootTile = HashToTile[instruction.Root];
385  // create neighbor tile
386  T tile = null;
387  tile = _factory.NewInstance<T>();
388  RegisterTile(tile, instruction.Tile);
389  // insert on the tile structure
390  var insert = HashToTile[instruction.Tile];
391  rootTile.InsertNeighbor(instruction.Direction, instruction.Rule, insert);
392  // register to internal structure
393  UpdateInstructions(rootTile, insert, instruction.Direction, instruction.Rule);
394  // update newer roots for next loop
395  newRoots.Add(instruction.Tile);
396 
397  tile.name = $"{tile.Controller.Tile.Index} -> {rootTile.Controller.Tile.Index}, {instruction.Direction} & {instruction.Rule}";
398 
399  // check if root is a level tile

401  {

403  insert.LevelId = levelId;

407  }
