65  {

71  List<KeyValuePair<GameLevelTile, GameLevelTile>> linksDone =
72  new List<KeyValuePair<GameLevelTile, GameLevelTile>>();
73 
74  foreach (var levelTile in LevelTiles)
75  {
76  var links = levelTile.Links.Links;
77 
78  // no linking for this tile
79  if (links == null || links.Count == 0) continue;
80 
81  var sourceState = _gameScoresHelper.State(levelTile.Map.Value);
82 
83  foreach (var levelTileLink in links)
84  {
85  // check if this link is already done
86  if (linksDone.Any((kv) =>
87  kv.Value == levelTile && kv.Key == levelTileLink.Level ||
88  kv.Key == levelTile && kv.Value == levelTileLink.Level))
89  {
90  continue;
91  }
92 
93  var linkState = _gameScoresHelper.State(levelTileLink.Level.Map.Value);
94 
95  // from locked to locked
96  if (linkState == LevelMapState.Locked || sourceState == LevelMapState.Locked)
97  {
98  InitializeLocked(levelTileLink, blocked);
99  }
100 
101  if (sourceState == LevelMapState.Completed && linkState == LevelMapState.Completed)
102  {
103  InitializeCompleted(levelTileLink, completed, levelTile);
104  }
105 
106  // from incomplete to incomplete
107  if (sourceState == LevelMapState.ToComplete && linkState == LevelMapState.ToComplete ||
108  sourceState == LevelMapState.Completed && linkState == LevelMapState.ToComplete)
109  {
110  // check if this tile was just completed
111  if (_bridge.Payload != null && _bridge.Payload.State == LevelMapState.ToComplete &&
112  levelTile.Map.Value.Id == _bridge.Payload.Level.Id)
113  {
114  InitializeRecentlyUnlocked(levelTileLink, justUnlocked, maxStars, levelTile);
115  }
116  else
117  {
118  InitializeIncomplete(levelTileLink, incomplete, levelTile);
119  }
120  }
121 
122  linksDone.Add(new KeyValuePair<GameLevelTile, GameLevelTile>(levelTile, levelTileLink.Level));
123  }
124  }
125 
126 
127  for (int i = 0; i < blocked.Count; i++)
128  {
129  var animator = blocked.ElementAt(i);
130  animator.SetFloat("Offset", (float) (i + 1) / blocked.Count);
131  animator.SetBool("IsBlocked", true);
132  }
133 
134  for (int i = 0; i < justUnlocked.Count; i++)
135  {
136  var animator = justUnlocked.ElementAt(i);
137  var pos = animator.transform.position;
138  var rot = animator.transform.rotation;
139  var sca = animator.transform.localScale;
140 
141  animator.SetFloat("Offset", (float) (i + 1) / justUnlocked.Count);
142  animator.SetBool("IsBlocked", true);
143 
144  // first stop blocked path anim
145  Observable.Timer(TimeSpan.FromSeconds(.1f)).Subscribe(_ => { },
146  () => { animator.SetBool("IsBlocked", false); }).AddTo(this);
147  // now animate appear
148  var time = .1f + (i + 1) * .3f;
149  Observable.Timer(TimeSpan.FromSeconds(time)).Subscribe(_ => { },
150  () =>
151  {
152  {
153  animator.SetTrigger("IsNowAvailable");
154  var meshRenderer = animator.GetComponentInChildren<MeshRenderer>();
155  meshRenderer.material = _starsColorHandler.GetMaterial(maxStars[animator]);
156  }
157  }).AddTo(this);
158 
159  // restore to original transform
160  Observable.Timer(TimeSpan.FromSeconds(time + .5f)).Subscribe(_ => { },
161  () =>
162  {
163  animator.enabled = false;
164  StartCoroutine(RestoreTransform(animator.transform, pos, rot, sca));
165  }).AddTo(this);

269  private IEnumerator RestoreTransform(Transform tr, Vector3 pos, Quaternion rot, Vector3 sca)
270  {
271  var t = 0f;
272 
273  var srPos = tr.position;
274  var srRot = tr.rotation;
275  var srSca = tr.localScale;
276  var time = .5f;
277 
278  while (t <= time)
279  {
280  tr.position = Vector3.Lerp(srPos, pos, t / time);
281  tr.rotation = Quaternion.Slerp(srRot, rot, t / time);
282  tr.localScale = Vector3.Lerp(srSca, sca, t / time);
283 
284  t += Time.deltaTime;
285  yield return null;
286  }
287 
288  tr.position = pos;
289  tr.rotation = rot;
290  tr.localScale = sca;
291  yield return null;
