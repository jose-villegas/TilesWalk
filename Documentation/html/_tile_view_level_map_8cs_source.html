300  public override void BuildTileMap<T>(LevelMap map)
301  {
302  Reset();
303  List<int> currentRoots = new List<int>();
304 
305  // create root tiles first
306  foreach (var rootTile in map.Roots)
307  {
308  T tile = null;
309  tile = _factory.NewInstance<T>();
310  RegisterTile(tile, rootTile.Key);
311  currentRoots.Add(rootTile.Key);
312  // set transform
313  tile.transform.position = rootTile.Position;
314  tile.transform.rotation = Quaternion.Euler(rootTile.Rotation);
315  // register root within the tile map
316  tile.Controller.Tile.Root = true;
317  _map.Roots.Add(rootTile);
318  // update indexes
319  UpdateIndexes(tile);
320  }
321 
322  var hasNewRootAvailable = true;
323 
324  // Now execute neighbor insertion logic
325  while (hasNewRootAvailable)
326  {
327  var newRoots = new List<int>();
328  hasNewRootAvailable = false;
329 
330  // now create all the tiles related to these roots
331  foreach (var root in currentRoots)
332  {
333  var related = map.Instructions.Where(x => x.Root == root).ToList();
334  var anyRelated = related.Any();
335  // ensure next loop since there is at least another tile parent
336  hasNewRootAvailable |= anyRelated;
337 
338  // no need to iterate if this tile is a leaf
339  if (!anyRelated) continue;
340 
341  foreach (var instruction in related)
342  {
343  var rootTile = HashToTile[instruction.Root];
344  // create neighbor instance
345  T tile = null;
346  tile = _factory.NewInstance<T>();
347  RegisterTile(tile, instruction.Tile);
348  // insert on the tile structure
349  var insert = HashToTile[instruction.Tile];
350  rootTile.InsertNeighbor(instruction.Direction, instruction.Rule, insert);
351  // update indexes
352  UpdateIndexes(tile);
353  // register to structure
354  UpdateInstructions(rootTile, insert, instruction.Direction, instruction.Rule);
355  // update newer roots for next loop
356  newRoots.Add(instruction.Tile);
357  }
358  }
359 
360  currentRoots = newRoots;
361  }
362 
363  _map.Id = map.Id;
364  _map.Target = map.Target;
365  _map.FinishCondition = map.FinishCondition;
366  _map.MapSize = map.MapSize;
367  _onLevelMapLoaded?.OnNext(_map);
368  }
