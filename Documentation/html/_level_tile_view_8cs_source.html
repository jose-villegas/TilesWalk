63  public void CheckSeparator()
64  {
65  var index = Controller.Tile.Index;
66 
67  for (int i = -1; i < 2; i++)
68  {
69  for (int j = -1; j < 2; j++)
70  {
71  for (int k = -1; k < 2; k++)
72  {
73  if (i == 0 && j == 0 && k == 0) continue;
74 
75  var checkIndex = new Vector3Int(index.x + i, index.y + j, index.z + k);
76 
77  if (_tileLevelMap.Indexes.TryGetValue(checkIndex, out var matching))
78  {
79  // check if any points up in the same direction
80  if (matching != null)
81  {
82  var matchUp = Math.Abs(Vector3.Dot(transform.up, matching.transform.up) - 1) <
83  Constants.Tolerance;
84  var notRelated = !Controller.Tile.Neighbors.ContainsValue(matching.Controller.Tile);
85 
86  if (!matchUp || !notRelated) continue;
87 
88  var direction = matching.transform.position - transform.position;
89  direction.Normalize();
90 
91  if (Math.Abs(Vector3.Dot(direction, transform.forward) - 1f) < Constants.Tolerance)
92  {
93  ParticleSystems.ParticleFX["S-North"].Play();
94  }
95  else if (Math.Abs(Vector3.Dot(direction, -transform.forward) - 1f) <
96  Constants.Tolerance)
97  {
98  ParticleSystems.ParticleFX["S-South"].Play();
99  }
100  else if (Math.Abs(Vector3.Dot(direction, transform.right) - 1f) < Constants.Tolerance)
101  {
102  ParticleSystems.ParticleFX["S-East"].Play();
103  }
104  else if (Math.Abs(Vector3.Dot(direction, -transform.right) - 1f) < Constants.Tolerance)
105  {
106  ParticleSystems.ParticleFX["S-West"].Play();
107  }
108  }
109  }
